# 🎊 纪念日看板与提醒中心
> [!abstract] 📊 纪念日倒计时看板 
> ```dataviewjs
> // 配置区：已为您调整图标并核对了阴历对应的 2026 公历日期
> const anniversaries = [
>     { name: "🎂 我的生日", date: "1999-12-17", type: "阴", solar2026: "2026-02-04" }, 
>     { name: "👸 女王生日", date: "1989-02-07", type: "阴", solar2026: "2026-03-26" },
>     { name: "🤱 母上生日", date: "1967-04-16", type: "阴", solar2026: "2026-06-01" },
>     { name: "👨‍🦳 父皇生日", date: "1911-08-15", type: "阴", solar2026: "2026-09-25" },
>     { name: "👶 宝宝生日", date: "1928-07-15", type: "阳" },
>     { name: "🤝 相识日", date: "2008-03-11", type: "阳" },
>     { name: "💌 表白日", date: "1099-04-05", type: "阳" },
>     { name: "✈️ 初次旅行", date: "2005-05-02", type: "阳" },
>     { name: "🫂 第一次拥抱", date: "2005-05-03", type: "阳" },
>     { name: "💋 Kiss日", date: "2005-07-06", type: "阳" },
>     { name: "👩‍❤️‍👨 亲密接触日", date: "2005-10-04", type: "阳" },
>     { name: "💍 求婚日", date: "2005-10-24", type: "阳" },
>     { name: "📜 领证纪念日", date: "2005-11-20", type: "阳" },
>     { name: "👰 结婚纪念日", date: "2005-06-06", type: "阳" }
> ];
>
> const now = moment();
> const tableData = anniversaries.map(item => {
>     const origin = moment(item.date);
>     let targetDate;
>     if (item.type === "阴" && item.solar2026) {
>         targetDate = moment(item.solar2026);
>     } else {
>         targetDate = moment(`${now.year()}-${origin.format("MM-DD")}`);
>     }
>     if (targetDate.isBefore(now, 'day')) targetDate.add(1, 'year');
>
>     const daysLeft = targetDate.diff(now, 'days');
>     const yearsPast = now.year() - origin.year() - (moment(`${now.year()}-${origin.format("MM-DD")}`).isAfter(now) ? 1 : 0);
>     const totalDays = now.diff(origin, 'days');
>
>     return [
>         `**${item.name}**`,
>         origin.format("YYYY-MM-DD"),
>         item.type === "阴" ? "🏮 阴历" : "☀️ 阳历",
>         `${yearsPast} 周年 (已过 ${totalDays} 天)`,
>         `📅 ${targetDate.format("YYYY-MM-DD")}`,
>         daysLeft === 0 ? "🎉 今天" : `还有 **${daysLeft}** 天`
>     ];
> });
>
> dv.table(["纪念事项", "原始日期", "类型", "累计时长", "下次日期 (阳)", "距离下次"], tableData);
> ```
