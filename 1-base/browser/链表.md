# 基本定义
```python
class Node：
	def __init__(self, prev, elment, next):
		self.val = elment
		self.next = next
		self.prev = prev

```

### 创建单链表
```
def createLinkedList(arr: 'List[int]') -> 'ListNode':
	if arr is None or len(arr) == 0:
		return None 
	
	head = ListNode(arr[0])
	cur = head
	for i in range(1, len(arr)):
		cur.next = ListNode(arr[i]):
		cur = cur.next
	return head 
```

### 增
- 头节点新增
```python
head = createLinkList([1,2,3,4,5])

p = head
newnode = ListNode(0)
newnode.next = p.next

p.next = newnode

```
- 尾节点新增
```python
p = head 
newnode = ListNode(0)

while p.next is not None:
	p = p.next

p.next = newnode 
```

## 双链表
```python
class DoubleListNode:
	def __init__(self, x):
		self.val = x
		self.next = None
		self.prev = None 

```
### 创建双链表
```python
def createDoubleLinkedList(arr: List[int]) -> Optional[DoubleLitNode]:
	if not arr:
		return None 
	
	head = DoubleListNode(arr[0])
	cur = head 
	
	for val in arr[1:]:
		new_node = DoubleListNode(val):
		cur.next = new_node
		new_node.prev = cur 
		cur = cur.next
	
	return head 
	

```

### 增
```python
head = createDoubleLinkedList([1,2,3,4,5,6])

p = head 
new_node = DoubleListNode(0)

new_node.next = p
p.prev = new_node
p = new_node
```
